1 goog.provide('ktree.GoogleEarth');
  2 
  3 goog.require('ktree.Utils');
  4 
  5 /**
  6 *	@fileoverview
  7 *	Implements the bridge between KarunaTree and the Google Earth API.
  8 *	This is the only place in the system where direct calls to the
  9 *	Earth API should be issued.
 10 */
 11 
 12 /**
 13 *	Create a new GoogleEarth object, attached to an initialized instance
 14 *	of the GoogleEarth API.
 15 *	@constructor
 16 */
 17 ktree.GoogleEarth = function(identifier) {
 18 	
 19 	/**
 20 	*	The Google Earth API instance associated with this object.
 21 	*	@private
 22 	*/
 23 	this.ge_;
 24 	
 25 	/**
 26 	*	A string to uniquely identify this GoogleEarth object
 27 	*	@private
 28 	*	@type {string}
 29 	*/
 30 	this.identifier_ = identifier;
 31 	
 32 	/**
 33 	*	A temporary reference to this *particular* GoogleEarth object needed to 
 34 	*	allow the initialization callback chain to reach this object again.
 35 	*/
 36 	var that = this;
 37 	
 38 	/**
 39 	*	Callback to begin initialization of the Google Earth API instance,
 40 	*	providing further callbacks for success and failure
 41 	*	@private	
 42 	*/
 43 	var initializeAPI_ = function() {
 44 		ktree.Utils.logFine('GoogleEarth <' + that.identifier_ + '> received its google.OnLoadCallback.');
 45 		google.earth.createInstance('earth', apiInitSuccess_, apiInitFailure_);
 46 	}
 47 	
 48 	/**
 49 	*	If the API initializes successfully, set up its basic parameters
 50 	*	@private
 51 	*	@param instance The initialized earth API instance
 52 	*/
 53 	var apiInitSuccess_ = function(instance) {
 54 		ktree.Utils.logFine('GoogleEarth <' + that.identifier_ + '> reports successful Earth API initialization.');
 55 		that.ge_ = instance;
 56 		that.ge_.getWindow().setVisibility(true);
 57 		that.ge_.getSun().setVisibility(false);
 58 		that.ge_.getOptions().setAtmosphereVisibility(true);
 59 		that.ge_.getNavigationControl().setVisibility(that.ge_.VISIBILITY_AUTO);
 60 	};
 61 
 62 	/**
 63 	*	If the API fails to initialize, provide an error message
 64 	*	@private
 65 	*	@param errorCode An error message from the Earth API
 66 	*/
 67 	var apiInitFailure_ = function(errorCode) {
 68 		alert("Oops! There was a problem initializing the Google Earth API.");
 69 		ktree.Utils.logSevere('GoogleEarth <' + that.identifer_ + '> reports that Earth API initialization failed.');
 70 		ktree.Utils.logSevere('API error code as follows:' + errorCode);
 71 	};
 72 	
 73 	// Finally! Start the process of setting up the Earth API
 74 	google.load("earth", "1");
 75 	google.setOnLoadCallback(initializeAPI_);
 76 }
