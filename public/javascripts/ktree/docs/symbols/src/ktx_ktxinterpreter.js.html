 16 /**
 17 *	//TODO Should the transmission to the World object happen here or in the KtxManager?
 18 *	
 19 *	Parse KTX from an argument KtxDataSource. If the World is ready, transmit the results for rendering.
 20 *	If the world is not yet ready, cache the results and retry transmission later
 21 *	@private
 22 *	@param {ktree.ktx.KtxDataSource} ktxDataSource The source of the KTX data to be parsed/transmitted
 23 */
 24 ktree.ktx.KtxInterpreter.prototype.parseKtx = function(dataSource, delay) {
 25 	var apiReady = this.world_.apiReady();
 26 	
 27 	//TODO Parse out KTX commands
 28 	var kmlString = goog.dom.xml.serialize(dataSource.getRootElement());
 29 	
 30 	if(apiReady) {
 31 		this.world_.parseKml(kmlString);
 32 	}
 33 	else {
 34 		this.delayedKtxTransmit_(kmlString);
 35 	}
 36 }
 37 
 38 /**
 39 *	@private
 40 */
 41 ktree.ktx.KtxInterpreter.prototype.delayedKtxTransmit_ = function(kmlString) {
 42 	ktree.debug.logInfo('The KtxInterpreter is delaying transmission of commands while GoogleEarth finishes initializing');
 43 	var target = this;
 44 	var conditionalDelay = new goog.async.ConditionalDelay(
 45 		function() {
 46 			return (target.world_.apiReady());
 47 		}
 48 	);
 49 	conditionalDelay.onFailure = function() {
 50 		ktree.debug.logError('The KtxInterpreter timed out waiting for GoogleEarth to initialize');
 51 	}
 52 	conditionalDelay.onSuccess = function() {
 53 		target.world_.parseKml(kmlString);
 54 		ktree.debug.logInfo('The KtxInterpreter reports that GoogleEarth finished initializing and accepted delayed commands');
 55 	}
 56 	conditionalDelay.start(100, 5000);
 57 }
