 32 /**
 33 *	Fade in a specified sound over a specified period of time. The sound does not have to
 34 *	have been previously loaded; requests to fade in unloaded sounds will result in attempt
 35 *	to load them.
 36 *	@param {string} soundID			The ID string of the sound to fade
 37 *	@param {string} relativePath	Path to the sound file, relative to the public root
 38 *	@param {integer} fadeDuration	(Optional) The duration of the fade in milliseconds
 39 */
 40 ktree.Soundscape.prototype.fadeIn = function(soundID, jsonParams) {
 41 	var params = this.mergeParamsWithDefaults_(jsonParams);
 42 	var sound = this.getSound_(soundID, params);
 43 	if (!sound) {
 44 		ktree.debug.logError('Soundscape.fadeIn could not find a sound with ID <' + soundID + '>');
 45 		return;
 46 	}
 47 	
 48 	ktree.debug.logGroup('Soundscape.fadeIn is fading in sound <' + soundID + '>');
 49 	var fadeCounter = 0;
 50 	var fadeDelay = new goog.async.ConditionalDelay(
 51 		function() {
 52 			if (fadeCounter < params.fade_in_duration) {
 53 				fadeCounter += ktree.sound.FADE_TIMESTEP;
 54 				sound.setVolume(params.volume*fadeCounter/params.fade_in_duration);
 55 				//ktree.debug.logInfo('Fade step <' + fadeCounter + '> : Sound volume: <' + sound.volume + '>');
 56 				return false;
 57 			}
 58 			if (fadeCounter >= params.fade_in_duration) {
 59 				sound.setVolume(params.volume)
 60 				//ktree.debug.logInfo('Fade step <' + fadeCounter + '> : Sound volume: <' + sound.volume + '>');
 61 				return true;
 62 			}
 63 		}
 64 	);
 65 	fadeDelay.onFailure = function() {
 66 		ktree.debug.logWarning('Soundscape.fadeIn failed to reach the end of the fade succesfully.');
 67 		sound.setVolume(params.volume);
 68 	}
 69 	fadeDelay.onSuccess = function() {
 70 		ktree.debug.logInfo('Fade in complete! Sound volume is: <' + sound.volume + '>');
 71 		ktree.debug.logGroupEnd();
 72 	}
 73 	
 74 	soundManager.play(soundID);
 75 	fadeDelay.start(ktree.sound.FADE_TIMESTEP, params.fade_in_duration + ktree.sound.FADE_DURATION_ERROR);
 76 }

 78 /**
 79 *	Fade out a specified sound over a specified period of time. Yields an error if the specified
 80 *	sound has not already been loaded.
 81 *	@param {string} soundID			The ID string of the sound to fade
 82 *	@param {integer} fadeDuration	(Optional) The duration of the fade in milliseconds
 83 */
 84 ktree.Soundscape.prototype.fadeOut = function(soundID, fadeDuration) {
 85 	var sound = soundManager.getSoundById(soundID);
 86 	if (!sound) {
 87 		ktree.debug.logError('Soundscape.fadeOut could not find a sound with ID <' + soundID + '>');
 88 		return;
 89 	}
 90 	if (!fadeDuration) {
 91 		fadeDuration = ktree.sound.DEFAULT_FADE_DURATION;
 92 	}
 93 	var numSteps = fadeDuration/ktree.sound.FADE_TIMESTEP;
 94 	var stepSize = sound.volume/numSteps;
 95 	
 96 	ktree.debug.logGroup('Soundscape.fadeOut is fading out sound <' + soundID + '>');
 97 	var fadeCounter = 0;
 98 	var fadeDelay = new goog.async.ConditionalDelay(
 99 		function() {
100 			if (fadeCounter < fadeDuration) {
101 				fadeCounter += ktree.sound.FADE_TIMESTEP;
102 				sound.setVolume(sound.volume-stepSize);
103 				//ktree.debug.logInfo('Fade step <' + fadeCounter + '> : Sound volume: <' + sound.volume + '>');
104 				return false;
105 			}
106 			if (fadeCounter >= fadeDuration) {
107 				sound.stop();
108 				//ktree.debug.logInfo('Fade step <' + fadeCounter + '> : Sound volume: <0>');
109 				return true;
110 			}
111 		}
112 	);
113 	fadeDelay.onFailure = function() {
114 		ktree.debug.logWarning('Soundscape.fadeOut failed to reach the end of the fade succesfully.');
115 		sound.stop();
116 	}
117 	fadeDelay.onSuccess = function() {
118 		ktree.debug.logInfo('Fade out complete!');
119 		ktree.debug.logGroupEnd();
120 	}
121 	fadeDelay.start(ktree.sound.FADE_TIMESTEP, fadeDuration + ktree.sound.FADE_DURATION_ERROR);
122 }

124 /**
125 *	Method used to queue callback functions that should execute once the SoundManager has finished 
126 *	initializing. Note that if the SoundManager is already initialized, the callback will be executed 
127 *	immediately.
128 *
129 *	Example: this method can be used to request a 'getSound' invocation when a page first loads, before
130 *	the SoundManager is fully initialized. The getSound request will be delayed (for up to five seconds)
131 *	until the SoundManager is ready to respond.
132 *	@public
133 *	@param {function} callback	The function that should be executed when the SoundManager is initialized.
134 */
135 ktree.Soundscape.prototype.whenReady = function(callback) {
136 	if (soundManager.supported()) {
137 		callback();
138 	}
139  	else {
140 		ktree.debug.logInfo('Soundscape.whenReady is delaying a callback while SoundManager initializes...');
141 		var readyDelay = new goog.async.ConditionalDelay(
142 			function() {
143 				return soundManager.supported()
144 			}
145 		);
146 		readyDelay.onFailure = function() {
147 			ktree.debug.logError('Soundscape.whenReady timed out while waiting for SoundManager to initialize');
148 		}
149 		readyDelay.onSuccess = function() {
150 			ktree.debug.logInfo('SoundManager is ready! Soundscape.whenReady is firing its callback.');
151 			callback();
152 		}
153 		readyDelay.start(100, 5000);
154 	}
155 }

157 /**
158 *	Retrieves the sound identified by a given sound ID. If the sound has not yet been loaded into the
159 *	SoundManager, this method will attempt to load a file with name soundID.mp3 from the default sound
160 *	directory. May return null if no sound could be matched to the ID.
161 *	@private
162 *	@param {string}		soundID		String identifier for the desired sound
163 *	@param {string} relativePath	Path to the sound file, relative to ktree.sound.SOUND_ROOT
164 *	@return {SMSound} 	sound		May be null if no sound could be matched to the argument ID
165 */
166 ktree.Soundscape.prototype.getSound_ = function(soundID, params) {
167 	var sound = soundManager.getSoundById(soundID);
168 	if (!sound) {
169 		sound = this.loadSound(soundID, params);
170 		if (!sound) {
171 			ktree.debug.logError('Soundscape.getSound could not find a sound with ID <' + soundID + '>');
172 			return null;
173 		}
174 	}
175 	return sound;
176 }

187 ktree.Soundscape.prototype.loadSound = function(soundID, params) {
188 	var smParams = {
189 		id: params.name,
190 		url: goog.string.buildString(ktree.sound.SOUND_ROOT, params.path),
191 		volume: params.volume,

193 			this.setPosition(params.start_position);
194 		}
195 	};
196 	return soundManager.createSound(smParams);
197 }

199 ktree.Soundscape.prototype.mergeParamsWithDefaults_ = function(jsonParams) {
200 	var params = jsonParams.sound;
201 	if (!params.volume) {
202 		params.volume = ktree.sound.DEFAULT_VOLUME;
203 	}
204 	if (!params.fade_in_duration) {
205 		params.fade_in_duration = ktree.sound.DEFAULT_FADE_DURATION;
206 	}
207 	return params;	
208 }

210 /**
211 *	TODO: This function is not currently working! soundManager.canPlayURL does not actually check for the existence
212 *	of a file at the URL -- it just checks that the URL is of valid format and file type.
213 *
214 *	Given an argument sound ID string, attempt to find and load a "matching" (i.e. named soundID.mp3) in the
215 *	server's default sound directory.
216 *	@private
217 *	@param {string}		soundID		ID string identifying the desired sound
218 *	@return {SMSound} 	sound		The sound matching the argument ID string. May be null if no match is found.
219 */
220 ktree.Soundscape.prototype.findAndLoadSound_ = function(soundID, targetVolume) {
221 	if (!targetVolume) {
222 		targetVolume = ktree.sound.DEFAULT_TARGET_VOLUME;
223 	}
224 	var music_url = goog.string.buildString(ktree.sound.MUSIC_DIR, soundID, '.mp3');
225 	if (soundManager.canPlayURL(music_url)) {
226 		this.targetVolumes_.set(soundID, targetVolume);
227 		return soundManager.createSound(soundID, music_url);
228 	}
229 	else {
230 		var ambient_url = goog.string.buildString(ktree.sound.AMBIENT_DIR, soundID, '.mp3');
231 		if (soundManager.canPlayURL(ambient_url)) {
232 			this.targetVolumes_.set(soundID, targetVolume);
233 			return soundManager.createSound(soundID, ambient_url);
234 		}
235 		else {
236 			ktree.debug.logError('Soundscape.findAndLoadSound could not locate a sound named <' + 
237 									soundID + '.mp3> in the ' + 
238 									ktree.sound.MUSIC_DIR + ' or ' +
239 									ktree.sound.FX_DIR + ' directories.');
240 			return null;
241 		}
242 	}
243 }
