 24 /**
 25 *	Create a Core object. This is the function called
 26 *	from HTML to setup the JS portion of our system.
 27 * 	@constructor
 28 */
 29 ktree.Core = function(debugOn) {
 30 	
 31 	if(debugOn) {
 32 		ktree.debug.startDebugger();
 33 	}
 34 
 35 	this.world_ = new ktree.GoogleEarth('Main');
 36 	this.kmlManager_ = new ktree.kml.KmlManager(this.world_);
 37 	
 38 	this.soundscape_ = new ktree.Soundscape();
 39 	
 40 	this.viewportSizeMonitor_ = new goog.dom.ViewportSizeMonitor();
 41 	goog.events.listen(this.viewportSizeMonitor_, goog.events.EventType.RESIZE, this.viewportResize_, false, this);
 42 	
 43 	var viewportSize = goog.dom.getViewportSize();
 44 	ktree.debug.logInfo('Core is setting up SplitPane view for a viewport of size: ' + viewportSize);
 45 	var lhs = new goog.ui.Component();
 46 	var rhs = new goog.ui.Component();
 47 	this.splitPane_ = new goog.ui.SplitPane(lhs, rhs, goog.ui.SplitPane.Orientation.HORIZONTAL);
 48 	//The initial size (for the left-half of the split) needs to be set before decorate is called
 49 	this.splitPane_.setInitialSize(viewportSize.width*0.7);
 50 	this.splitPane_.decorate(goog.dom.getElement('splitter'));
 51 	this.splitPane_.setHandleSize(15);
 52 	var scaledViewport = new goog.math.Size(viewportSize.width, viewportSize.height*0.92);
 53 	this.splitPane_.setSize(scaledViewport);
 54 	this.splitPane_.setContinuousResize(true);
 55 }
 56 
 57 ktree.Core.prototype.viewportResize_ = function(e) {
 58 	var newSize = this.viewportSizeMonitor_.getSize();
 59 	ktree.debug.logInfo('Core is resizing the SplitPane view for new viewport dimensions: ' + newSize);
 60 	var newScaledSize =new goog.math.Size(newSize.width, newSize.height*0.92);
 61 	this.splitPane_.setSize(newScaledSize);	
 62 }
 63 
 64 ktree.Core.prototype.soundscape = function() {
 65 	return this.soundscape_;
 66 }

 68 ktree.Core.prototype.loadKml = function(uri, dsName) {
