  8 /**

 16 /**
 17 *	Start the Main debug log and attach it to a window
 18 *	
 19 */
 20 ktree.debug.startDebugger = function() {
 21 	
 22 	//Uncomment this version to use a separate debug window
 23 	//
 24 	//var debugWindow = new goog.debug.FancyWindow('KarunaTree');
 25 	//debugWindow.setEnabled(true);
 26 	//debugWindow.init();
 27 	
 28 	var debugConsole = new goog.debug.Console();
 29 	debugConsole.setCapturing(true);
 30 
 31 	ktree.debug.mainLogger_ = goog.debug.Logger.getLogger('Main');
 32 	ktree.debug.mainLogger_.setLevel(goog.debug.Logger.Level.FINE);
 33 	ktree.debug.logFine('KTree is running in debug mode');
 34 }
 35 
 36 
 37 ktree.debug.logFine = function(message) {
 38 	//console.info(message);
 39 	ktree.debug.mainLogger_.fine(message);
 40 }
 41 
 42 ktree.debug.logInfo = function(message) {
 43 	//console.info(message);
 44 	ktree.debug.mainLogger_.info(message);
 45 }
 46 
 47 ktree.debug.logWarning = function(message) {
 48 	//console.warning(message);
 49 	ktree.debug.mainLogger_.warning(message);
 50 }
 51 
 52 ktree.debug.logError = function(message) {
 53 	//console.error(message)
 54 	ktree.debug.mainLogger_.error(message);
 55 }
 56 
 57 ktree.debug.logElement = function(element, message) {
 58 	if(message) ktree.debug.mainLogger_.info(message);
 59 	console.dirxml(element);
 60 }
 61 
 62 ktree.debug.logProperties = function(object, message) {
 63 	if(message) ktree.debug.mainLogger_.info(message);
 64 	console.dir(object);
 65 }
 66 
 67 ktree.debug.logGroup = function(message) {
 68 	console.group(message);
 69 }
 70 
 71 ktree.debug.logGroupEnd = function() {
 72 	console.groupEnd();
 73 }
