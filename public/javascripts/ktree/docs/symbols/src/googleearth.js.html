 20 /**
 21 *	Create a new GoogleEarth object, attached to an initialized instance
 22 *	of the GoogleEarth API.
 23 *	@constructor
 24 */
 25 ktree.GoogleEarth = function(identifier) {
 26 	
 27 	/**
 28 	*	A string to uniquely identify this GoogleEarth object
 29 	*	@private
 30 	*	@type {string}
 31 	*/
 32 	this.identifier_ = identifier;
 33 	
 34 	/**
 35 	*	The Google Earth API instance associated with this object.
 36 	*	@private
 37 	*/
 38 	this.ge_;
 39 		
 40 	/**
 41 	*	Flag indicating whether or not the Google Earth API instance is
 42 	*	initialized and ready for commands
 43 	*	@private
 44 	*	@type {boolean}
 45 	*/
 46 	this.geReady_ = false;
 47 
 48 	
 49 	/**
 50 	*	A temporary reference to this *particular* GoogleEarth object needed to 
 51 	*	allow the initialization callback chain to reach this object again.
 52 	*	@private
 53 	*/
 54 	var target_ = this;
 55 		
 56 	/**
 57 	*	@private
 58 	*	@type {ktree.AnimationProvider}
 59 	*/
 60 	this.animationProvider_ = null;
 61 	
 62 	/**
 63 	*	Callback to begin initialization of the Google Earth API instance,
 64 	*	providing further callbacks for success and failure
 65 	*	@private	
 66 	*/
 67 	var initializeAPI_ = function() {
 68 		ktree.debug.logFine('GoogleEarth <' + target_.identifier_ + '> received its google.OnLoadCallback.');
 69 		google.earth.createInstance('earth', apiInitSuccess_, apiInitFailure_);
 70 	}
 71 	
 72 	/**
 73 	*	If the API initializes successfully, set up its basic parameters
 74 	*	@private
 75 	*	@param instance The initialized earth API instance
 76 	*/
 77 	var apiInitSuccess_ = function(instance) {
 78 		ktree.debug.logFine('GoogleEarth <' + target_.identifier_ + '> reports successful Earth API initialization.');
 79 		target_.ge_ = instance;
 80 		target_.geReady_ = true;
 81 		target_.ge_.getWindow().setVisibility(true);
 82 		target_.ge_.getSun().setVisibility(false);
 83 		target_.ge_.getOptions().setAtmosphereVisibility(true);
 84 		target_.ge_.getNavigationControl().setVisibility(target_.ge_.VISIBILITY_AUTO);
 85 	};
 86 
 87 	/**
 88 	*	If the API fails to initialize, provide an error message
 89 	*	@private
 90 	*	@param errorCode An error message from the Earth API
 91 	*/
 92 	var apiInitFailure_ = function(errorCode) {
 93 		alert("Oops! There was a problem initializing the Google Earth API.");
 94 		ktree.debug.logError('GoogleEarth <' + target_.identifier_ + '> reports that Earth API initialization failed.');
 95 		ktree.debug.logError('API error code as follows:' + errorCode);
 96 	};
 97 	
 98 	// Finally! Start the process of setting up the Earth API
 99 	google.load("earth", "1");
100 	google.setOnLoadCallback(initializeAPI_);
101 }
102 
103 /**
104 *	Returns whether or not the Google Earth API is initialized and ready for commands
105 *	@return {boolean}
106 */
107 ktree.GoogleEarth.prototype.apiReady = function() {
108 	return this.geReady_;
109 }

111 /**
112 *	Adds the features in an argument KML string to the GE instance. If the KML string defines
113 *	an AbstractView (e.g. a Camera or a LookAt), the GE instance's view is set to the
114 *	AbstractView
115 *	@param {string} kmlString		A string of valid KML
116 */
117 ktree.GoogleEarth.prototype.parseKml = function(kmlString) {
118 	var kmlObject = this.ge_.parseKml(kmlString);
119 	if (kmlObject.getAbstractView() !== null) {
120 		this.ge_.getView().setAbstractView(kmlObject.getAbstractView());
121 	}
122 	this.ge_.getFeatures().appendChild(kmlObject);
123 }
124 
125 ktree.GoogleEarth.prototype.fetchKml = function(kmlUrl) {
126 	ktree.debug.logFine('GoogleEarth <' + this.identifier_ + '> is fetching KML from from URL <' + kmlUrl + '>');
127 	var callbackTarget = this;
128 	google.earth.fetchKml(this.ge_, kmlUrl, function(kmlObject) {callbackTarget.addKml(kmlObject)});
129 }
130 
131 ktree.GoogleEarth.prototype.addKml = function(kmlObject) {
132 	ktree.debug.logFine('GoogleEarth <' + this.identifier_ + '> is parsing a new KML Object...');
133 	if (kmlObject) {
134 		
135 		if('getFeatures' in kmlObject) {
136 			ktree.debug.logFine('Here is the KML Object: ' + kmlObject.getKml());
137 			var featureList = kmlObject.getFeatures().getChildNodes();
138 			ktree.debug.logFine('KML object has features: ' + featureList.getLength());
139 			var i;
140 			for(i = 0; i < featureList.getLength(); i++) {
141 				ktree.debug.logFine('Loop: ' +i);
142 				var feature = featureList.item(i);
143 				ktree.debug.logFine('Feature:' +feature.getType());
144 			}
145 		}
146 		
147 		if (kmlObject.getAbstractView() !== null) {
148 			this.ge_.getView().setAbstractView(kmlObject.getAbstractView());
149 		}
150 		this.ge_.getFeatures().appendChild(kmlObject);
151 	}
152 	
153 }

155 ktree.GoogleEarth.prototype.moveCameraTo = function() {
156 	// Get the current view
157 	var lookAt = this.ge_.getView().copyAsLookAt(this.ge_.ALTITUDE_RELATIVE_TO_GROUND);
158 	
159 	// Set lat and long as desired
160 	lookAt.setLatitude(0);
161 	lookAt.setLongitude(0);
162 	
163 	// Update the view
164 	this.ge_.getView().setAbstractView(lookAt);
165 }
166 
167 ktree.GoogleEarth.prototype.moveCameraRelative = function() {
168 	var camera = this.ge_.getView().copyAsCamera(this.ge_.ALTITUDE_RELATIVE_TO_GROUND);
169 	
170 	camera.setLatitude(camera.getLatitude() + 0);
171 	camera.setLongitude(camera.getLongitude() + 0.1);
172 	
173 	this.ge_.getView().setAbstractView(camera);
174 }

176 /**
177 *	@param {ktree.AnimationProvider | null} newAnimationProvider
178 */
179 ktree.GoogleEarth.prototype.startAnimation = function(newAnimationProvider) {
180 	if(!this.animationRunning_) {
181 		this.animationRunning_ = true;
182 		if(goog.isDef(newAnimationProvider)) {
183 			this.setAnimationProvider(newAnimationProvider);
184 		}
185 		if (!goog.isNull(this.animationProvider_)) {
186 			ktree.debug.logInfo('GoogleEarth is starting the AnimationProvider <' + this.animationProvider_.getName() + '>.');
187 			google.earth.addEventListener(this.ge_, 'frameend', this.animationProvider_.provide);
188 			this.animationProvider_.initialize();
189 			this.animationProvider_.provide();
190 		}
191 		else {
192 			ktree.debug.logWarning('GoogleEarth was asked to start an animation without an AnimationProvider.');
193 		}
194 	}
195 	else {
196 		ktree.debug.logWarning('GoogleEarth was asked to start an animation while another was still running.');
197 	}
198 	
199 }
200 
201 ktree.GoogleEarth.prototype.startAnimationCallback = function() {
202 	this.startAnimation();
203 }
204 
205 ktree.GoogleEarth.prototype.stopAnimation = function(tickProvider) {
206 	if(this.animationRunning_) {
207 		ktree.debug.logInfo('GoogleEarth is stopping the AnimationProvider <' + this.animationProvider_.getName() + '>.');
208 		google.earth.removeEventListener(this.ge_, 'frameend', this.animationProvider_.provide);
209 		this.animationRunning_ = false;
210 		this.animationProvider_.cleanup();
211 	}
212 	else {
213 		ktree.debug.logWarning('GoogleEarth was asked to stop an animation when none was running.');
214 	}
215 }

221 ktree.GoogleEarth.prototype.setAnimationProvider = function(newAnimationProvider) {
222 	ktree.debug.logInfo('GoogleEarth is switching to the AnimationProvider <' + newAnimationProvider.getName() + '>.');
223 	this.animationProvider_ = newAnimationProvider;
224 }

227 	//alert('Google Earth <' + this.identifier_ + "> is making a camera button");

235 	b.render(goog.dom.$('button'));
236 	var target = this;
237 	
238 	/**

240 	*/

244 		},
245

249 			target.ge_.getOptions().setStatusBarVisibility(true);
250 		},
251 		
252 		provide: function() {
253 			target.moveCameraRelative();
254 		},
255 		
256 		cleanup: function() {
257 			
258 		}
259 	};
260 	
261 	this.setAnimationProvider(orbit);
262 	goog.events.listen(b, goog.ui.Component.EventType.ACTION, this.startAnimationCallback, false, this);
