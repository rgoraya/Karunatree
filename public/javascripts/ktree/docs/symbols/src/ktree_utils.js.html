 22 /**
 23 *	Start the Main debug log and attach it to a window
 24 *	
 25 */
 26 ktree.Utils.startDebugger = function() {
 27 	//Create the debug window
 28 	var debugWindow = new goog.debug.FancyWindow('KarunaTree');
 29 	debugWindow.setEnabled(true);
 30 	debugWindow.init();
 31 
 32 	//Create a logger to display messages "FINE" and above
 33 	var logger = goog.debug.Logger.getLogger('Main');
 34 	logger.setLevel(goog.debug.Logger.Level.FINE);
 35 	ktree.Utils.logFine('Running in debug mode');
 36 }
 37 
 38 /**
 39 *	Log message with level FINE to the Main debug log.
 40 *	Note this is an example of a public static method.
 41 * @private
 42 */
 43 ktree.Utils.logFine = function(message) {
 44 	var logger = goog.debug.Logger.getLogger('Main');
 45 	logger.fine(message);
 46 }
 47 
 48 /**
 49 *	Log message with level SEVERE to the Main debug log.
 50 */
 51 ktree.Utils.logSevere = function(message) {
 52 	var logger = goog.debug.Logger.getLogger('Main');
 53 	logger.severe(message);
 54 }
