 29 /**
 30 *	Create a new GoogleEarth object, attached to an initialized instance
 31 *	of the GoogleEarth API.
 32 *
 33 *	@constructor
 34 */
 35 ktree.GoogleEarth = function(identifier) {
 36 	
 37 	/**
 38 	*	Used to disable fly-to-speed commands during debugging
 39 	*/
 40 	this.ignoreFlyToSpeed_ = false;
 41 	
 42 	/**
 43 	*	A string to uniquely identify this GoogleEarth object
 44 	*	@private
 45 	*	@type {string}
 46 	*/
 47 	this.identifier_ = identifier;
 48 	
 49 	/**
 50 	*	The Google Earth API instance associated with this object.
 51 	*	@private
 52 	*/
 53 	this.ge_;
 54 		
 55 	/**
 56 	*	Flag indicating whether or not the Google Earth API instance is
 57 	*	initialized and ready for commands
 58 	*	@private
 59 	*	@type {boolean}
 60 	*/
 61 	this.geReady_ = false;
 62 	
 63 	/**
 64 	*	Object used to send XHR requests back to the KarunaTree server
 65 	*	@private
 66 	*	@type {goog.net.XhrIo}
 67 	*/
 68 	
 69 	this.xhr_ = new goog.net.XhrIo();
 70 
 71 	/**
 72 	*	A temporary reference to this *particular* GoogleEarth object needed to 
 73 	*	allow the initialization callback chain to reach this object again.
 74 	*	@private
 75 	*/
 76 	var target_ = this;
 77 		
 78 	/**
 79 	*	Callback to begin initialization of the Google Earth API instance,
 80 	*	providing further callbacks for success and failure
 81 	*	@private	
 82 	*/
 83 	var initializeAPI_ = function() {
 84 		ktree.debug.logFine('GoogleEarth <' + target_.identifier_ + '> received its google.OnLoadCallback.');
 85 		google.earth.createInstance('earth', apiInitSuccess_, apiInitFailure_);
 86 	}
 87 	
 88 	/**
 89 	*	If the API initializes successfully, set up its basic parameters
 90 	*	@private
 91 	*	@param instance The initialized earth API instance
 92 	*/
 93 	var apiInitSuccess_ = function(instance) {
 94 		ktree.debug.logFine('GoogleEarth <' + target_.identifier_ + '> reports successful Earth API initialization.');
 95 		target_.ge_ = instance;
 96 		target_.geReady_ = true;
 97 		target_.ge_.getWindow().setVisibility(true);
 98 		target_.ge_.getSun().setVisibility(false);
 99 		target_.ge_.getOptions().setAtmosphereVisibility(true);
100 		target_.ge_.getNavigationControl().setVisibility(target_.ge_.VISIBILITY_AUTO);
101 	};
102 
103 	/**
104 	*	If the API fails to initialize, provide an error message
105 	*	@private
106 	*	@param errorCode An error message from the Earth API
107 	*/
108 	var apiInitFailure_ = function(errorCode) {
109 		ktree.debug.logError('GoogleEarth <' + target_.identifier_ + '> reports that Earth API initialization failed.');
110 		ktree.debug.logError('API error code as follows:' + errorCode);
111 	};
112 	
113 	// Finally! Start the process of setting up the Earth API
114 	google.load("earth", "1");
115 	google.setOnLoadCallback(initializeAPI_);
116 }

150 /**
151 *	TODO Need to remove hard-coded reference to Chapter 1
152 *
153 *	Adds the contents of a KML string to a specified parent KmlContainer within
154 *	the GE plugin. If no parent is specified, the KML is added to the top-level
155 *	of the plugin itself. Note that if the KML string defines a Camera or a LookAt, 
156 *	the plugin's current view is set accordingly.
157 *	@public
158 *	@param {string} kmlString		A string of valid KML
159 *	@param {string} parentName		Optional. The name of a KmlContainer to which the new KML should be parented
160 */
161 ktree.GoogleEarth.prototype.addKml = function(kmlString, parentName) {
162 	var kmlObject = this.kmlStringToObject_(kmlString);
163 	
164 	var parentContainer;
165 	if (parentName) {
166 		parentContainer = this.tryRetrievingKmlFeature_(parentName);
167 		if (!parentContainer) {
168 			parentContainer = this.createKmlFolder_("Chapter 1");
169 		}
170 	}
171 	else {
172 		// If no parent container is specified, parent the new KML to the top-level of the plugin
173 		parentContainer = this.ge_;
174 	}
175 	
176 	parentContainer.getFeatures().appendChild(kmlObject);
177 	
178 	this.runKtxCommandsForObject_(kmlObject, true);
179 	var objectView = kmlObject.getAbstractView();
180 	if (objectView) {
181 		this.updateView_(objectView);
182 	}
183 	
184 	this.serializeKmlToServer_();
185 }

188 /**
189 *	Set the camera's fly-to speed
190 *	@public
191 *	@param {float} speed	The desired fly-to speed. Accepts a float from 0.0 to 5.0, inclusive.
192 */
193 ktree.GoogleEarth.prototype.setFlyToSpeed = function(speed) {
194 	if (this.ignoreFlyToSpeed_) {
195 		return;
196 	}
197 	if (speed == ktree.kml.constants.SPEED_TELEPORT) {
198 		speed = this.ge_.SPEED_TELEPORT;
199 	}
200 	if (speed < 0.0 || speed > 5.0) {
201 		ktree.debug.logWarning("GoogleEarth received an invalied flyToSpeed request of <" + speed + ">. Camera speed is limited to 0.0 to 5.0, inclusive.");
202 		return;
203 	}
204 	this.ge_.getOptions().setFlyToSpeed(speed);
205 }

227 /**
228 *	Recurrent function parses through a KmlObject and all of its children, running any KTX commands
229 *	that have been cached for any of the nodes. Optionally uses the abstract view associated
230 *	with the argument KmlObject to update the current view.
231 *	@private
232 *	@param {KmlObject} kmlObject		The KmlObject to process
233 *	@param {boolean} shouldUpdateView	If true, attempts to use the abstract view associated with
234 *										kmlObject to update the plugin's view
235 */
236 ktree.GoogleEarth.prototype.runKtxCommandsForObject_ = function(kmlObject, shouldUpdateView) {
237 	var objectType = kmlObject.getType();
238 	
239 	// The contents of containers (KmlFolders and KmlDocuments) need to be recursively post-processed
240 	if (this.typeIsContainer_(objectType)) {
241 		var childObjectList = kmlObject.getFeatures().getChildNodes();
242 		var numChildren = childObjectList.getLength();
243 		for (var i = 0; i < numChildren; i++) {
244 			this.runKtxCommandsForObject_(childObjectList.item(i), shouldUpdateView);
245 		}
246 	}
247 	
248 	// Placemarks need to have click handlers installed
249 	else if (this.typeIsClickable_(objectType)) {
250 		var illustrationName = this.ktxCache_.illustrationForPlacemark(kmlObject.getName());
251 		this.createIllustrationClickHandler_(kmlObject, illustrationName);
252 	}
253 }

256 /**
257 *	Retrieve a KmlFeature object from the plugin. Raises an error if the requested feature cannot
258 *	be found. For situations where this error is undesirable, use the alternative method 
259 *	tryRetrievingKmlFeature_().
260 *	@see {ktree.GoogleEarth#tryRetrievingKmlFeature_}
261 *	@private
262 *	@param {string} featureName				The name (ID) of the KmlFeature to retrieve
263 *	@param {KmlContainer} rootContainer		Optional. The KmlContainer to use as the root of the search. If no argument is specified, the plugin root is used.
264 *	@returns {KmlFeature}					The named KmlFeature. Guaranteed not be null.
265 */
266 ktree.GoogleEarth.prototype.retrieveKmlFeature_ = function(featureName, rootContainer) {
267 	var kmlFeature = this.tryRetrievingKmlFeature_(featureName, rootContainer);
268 	if (!kmlFeature) {
269 		ktree.debug.logError("GoogleEarth couldn't retrieve the KmlFeature <" + featureName +"> when it was expected to exist.");
270 	}
271 	return kmlFeature;
272 }

275 /**
276 *	Attempt to retrieve a KmlFeature object from the plugin. May return a null result if the feature
277 *	cannot be found. For situations where a null result would be undesirable, use the alternative method
278 *	retrieveKmlFeature_().
279 *	@see {ktree.GoogleEarth#retrieveKmlFeature_}
280 *	@private
281 *	@param {string} featureName				The name (ID) of the KmlFeature to retrieve
282 *	@param {KmlContainer} rootContainer		Optional. The KmlContainer to use as the root of the search. If no argument is 
283 *											specified, the plugin root is used.
284 *	@return {KmlFeature}					The named KmlFeature. May be null if no matching feature can be found.
285 */
286 ktree.GoogleEarth.prototype.tryRetrievingKmlFeature_ = function(featureName, rootContainer) {
287 	
288 	ktree.debug.logFine("Searching for KmlFeature <" + featureName + ">...");
289 	if (!rootContainer) {
290 		rootContainer = this.ge_;
291 		ktree.debug.logFine("...Starting search in <Plugin Root>.");
292 	}
293 	else {
294 		ktree.debug.logFine("...Starting search in <" + rootContainer.getId() + ">");
295 	}
296 	
297 	var kmlObjectList = rootContainer.getFeatures().getChildNodes();
298 	ktree.debug.logFine("...<" + kmlObjectList.getLength() + "> child nodes to search...");
299 	
300 	for (var i = 0; i < kmlObjectList.getLength(); i++) {
301 		var kmlObject = kmlObjectList.item(i);
302 		
303 		if (kmlObject.getId() == featureName) {
304 			ktree.debug.logFine("...Found KmlFeature <" + featureName + ">.");
305 			return kmlObject;
306 		}
307 		
308 		if (kmlObject.getType() == "KmlFolder") {
309 			ktree.debug.logFine("...Recurring into KmlFolder <" + kmlObject.getId() + ">");
310 			return this.tryRetrievingKmlFeature_(featureName, kmlObject)
311 		}
312 	}
313 	
314 	// If we reach this point, the desired KmlFeature was not found
315 	ktree.debug.logFine("...Couldn't find any KmlFeature named <" + featureName + ">.")
316 	return null;
317 }

320 /**
321 *	Attempts to load KML data for the specified scene by retrieving it from the plugin's
322 *	internal cache. If successful, will also update the plugin's view using the retrieved
323 *	KML.
324 *	@private
325 *	@param {string} sceneName		The name of the scene to try and load
326 *	@returns {boolean}				Could KML for the specified scene be retrieved from the plugin?
327 */
328 ktree.GoogleEarth.prototype.tryLoadingKmlForScene = function(sceneName) {
329 	var sceneDocument = this.tryRetrievingKmlFeature_(sceneName);
330 	if (!sceneDocument) {
331 		return false;
332 	}
333 	
334 	var sceneView = sceneDocument.getAbstractView();
335 	if (sceneView) {
336 		this.updateView_(sceneView);
337 		return true;
338 	}
339 }

361 /**
362 *	Updates the plugin's current view using an argument abstract view. Will also set the camera
363 *	fly-to speed appropriately if there is a KTX parameter cached for the view.
364 *	@private
365 *	@param {KmlAbstractView} abstractView	The abstract view to use for updating
366 */
367 ktree.GoogleEarth.prototype.updateView_ = function(abstractView) {
368 	var flyToSpeed = this.ktxCache_.flyToSpeedForView(abstractView.getId());
369 	if (flyToSpeed) {
370 		this.setFlyToSpeed(flyToSpeed);
371 	}
372 	this.ge_.getView().setAbstractView(abstractView);
373 }

376 /**
377 *	Safe method for attempting to convert a KML string to a KmlObject. If the KML string is
378 *	well-formed, the resulting KmlObject is returned; otherwise an error is raised. 
379 *	@private
380 *	@params {string} kmlString		The KML string to convert to an object
381 *	@returns {KmlObject}
382 */
383 ktree.GoogleEarth.prototype.kmlStringToObject_ = function(kmlString) {
384 	try {
385 		var kmlObject = this.ge_.parseKml(kmlString);
386 	}
387 	catch (error) {
388 		ktree.debug.logError("Adding KML failed. GoogleEarth was unable to parse the KML string: " + kmlString);
389 	}
390 	return kmlObject;
391 }

394 /**
395 *	Adds a new KmlFolder object to the GoogleEarth plugin's internal data store and
396 *	returns a reference to it
397 *	@private
398 *	@param {string} folderName		The string ID of the new KmlFolder
399 *	@param {string} parentName		The string ID of the intended parent container. If null, the new folder is 
400 *									added to the top-level of the plugin.
401 *	@return {KmlFolder}				A pointer to the created KmlFolder
402 */
403 ktree.GoogleEarth.prototype.createKmlFolder_ = function(folderName, parentName) {
404 	var folder = this.ge_.createFolder(folderName);
405 	
406 	if (!parentName) {
407 		ktree.debug.logFine('Installing KmlFolder <' + folderName + '> with plugin root as parent.');
408 		this.ge_.getFeatures().appendChild(folder);
409 	}
410 
411 	else {
412 		ktree.debug.logFine('Installing KmlFolder <' + folderName + '> into container <' + parentName + '>');
413 		var parentContainer = this.retrieveKmlFeature_(parentName);
414 		parentContainer.getFeatures().appendChild(folder);
415 	}
416 	
417 	return folder;
418 }

468 ktree.GoogleEarth.prototype.moveCameraTo = function() {
469 	// Get the current view
470 	var lookAt = this.ge_.getView().copyAsLookAt(this.ge_.ALTITUDE_RELATIVE_TO_GROUND);
471 	
472 	// Set lat and long as desired

474 	lookAt.setLongitude(0);
475

484 	var camera = this.ge_.getView().copyAsCamera(this.ge_.ALTITUDE_RELATIVE_TO_GROUND);

493 /**
494 *	@ignore
495 *	@param {ktree.AnimationProvider | null} newAnimationProvider
496 */
497 ktree.GoogleEarth.prototype.startAnimation = function(newAnimationProvider) {
498 	if(!this.animationRunning_) {
499 		this.animationRunning_ = true;
500 		if(goog.isDef(newAnimationProvider)) {
501 			this.setAnimationProvider(newAnimationProvider);
502 		}
503 		if (!goog.isNull(this.animationProvider_)) {
504 			ktree.debug.logInfo('GoogleEarth is starting the AnimationProvider <' + this.animationProvider_.getName() + '>.');
505 			google.earth.addEventListener(this.ge_, 'frameend', this.animationProvider_.provide);
506 			this.animationProvider_.initialize();
507 			this.animationProvider_.provide();
508 		}
509 		else {
510 			ktree.debug.logWarning('GoogleEarth was asked to start an animation without an AnimationProvider.');
511 		}
512 	}
513 	else {
514 		ktree.debug.logWarning('GoogleEarth was asked to start an animation while another was still running.');
515 	}
516 	
517 }

531 		ktree.debug.logInfo('GoogleEarth is stopping the AnimationProvider <' + this.animationProvider_.getName() + '>.');
532 		google.earth.removeEventListener(this.ge_, 'frameend', this.animationProvider_.provide);
533 		this.animationRunning_ = false;
534 		this.animationProvider_.cleanup();
535 	}
536 	else {
537 		ktree.debug.logWarning('GoogleEarth was asked to stop an animation when none was running.');
538 	}
