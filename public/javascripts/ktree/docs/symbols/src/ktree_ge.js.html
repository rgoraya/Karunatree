1 goog.provide('ktree.world.ge');
  2 
  3 goog.require('ktree.Utils');
  4 
  5 ktree.world.ge.initialize = function() {
  6 	ktree.world.ge.loadAPI();
  7 };
  8 
  9 //Load the API
 10 ktree.world.ge.loadAPI = function() {
 11 	google.load("earth", "1");
 12 	google.setOnLoadCallback(this.initializeAPI);
 13 };
 14 
 15 //Initialize the GE instance, with callbacks
 16 ktree.world.ge.initializeAPI = function() {
 17 	google.earth.createInstance('earth', ktree.world.ge.apiInitSuccess, ktree.world.ge.apiInitFailure);
 18 };
 19 
 20 //Callback for succesful initialization
 21 ktree.world.ge.apiInitSuccess = function(instance) {
 22 	ktree.Utils.logFine('Google Earth API initialized successfully');
 23 	ktree.Utils.startDebugger();
 24 	ge = instance;
 25 	ge.getWindow().setVisibility(true);
 26 	ge.getSun().setVisibility(false);
 27 	ge.getOptions().setAtmosphereVisibility(true);
 28 	ge.getNavigationControl().setVisibility(ge.VISIBILITY_AUTO);
 29 };
 30 
 31 //Callback for failed initialization
 32 ktree.world.ge.apiInitFailure = function(errorCode) {
 33 	alert("Oops! There was a problem connecting to the Google Earth API.");
 34 	ktree.Utils.logSevere('The Google Earth API failed to initialize.');
 35 };
